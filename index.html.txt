    <py-script>
        from js import document
        from pyodide.ffi import create_proxy
        import yfinance as yf

        def analisar_no_clique(event):
            # 1. Pega o texto que o usuário digitou na caixinha
            ticker = document.getElementById("ticker-input").value.strip().upper()
            output_div = document.getElementById("output")
            
            if not ticker:
                output_div.innerHTML = "⚠️ Por favor, digite o código de uma ação!"
                return

            output_div.innerHTML = f"🔄 Buscando dados de {ticker} na Bolsa..."

            try:
                # 2. Roda a sua lógica autônoma
                acao = yf.Ticker(ticker)
                historico = acao.history(period="1mo")
                
                preco_maximo = historico['High'].max()
                preco_minimo = historico['Low'].min()
                oscilacao = preco_maximo - preco_minimo
                
                if oscilacao > 4.00:
                    perfil = "🚀 ALTO RISCO (Alta variação)"
                else:
                    perfil = "🛡️ BAIXO RISCO (Estável)"
                
                # 3. Devolve o resultado direto na tela para o seu amigo ver
                output_div.innerHTML = f"<strong>Ação:</strong> {ticker}<br><strong>Oscilação Mensal:</strong> R$ {oscilacao:.2f}<br><strong>Perfil:</strong> {perfil}"
            
            except Exception as e:
                output_div.innerHTML = "❌ Erro ao buscar ativo. Verifique o código (Ex: PETR4.SA)"

        # Vincula o clique do botão à nossa função Python
        botao = document.getElementById("btn-analisar")
        botao.addEventListener("click", create_proxy(analisar_no_clique))
    </py-script>
